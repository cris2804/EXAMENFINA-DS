package org.example;

public class Lluvia {

    //Nuevo Constructor de la clase Lluvia
     Lluvia(double volumen) {
        Lluvia.volumen = volumen;
    }

     private static ExternalVolumenService externalvolumen; //atributo de la clase CantidadLluvia
